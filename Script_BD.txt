SCRIPT PARA CREAR TABLAS DEL PROYECTO DE RESERVAS HOTELERAS

-- Tabla para el Microservicio de Usuarios
CREATE TABLE USUARIOS (
    ID_USUARIO    NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    USERNAME      VARCHAR2(20) NOT NULL,
    PASSWORD      VARCHAR2(255) NOT NULL, 
    ROL           VARCHAR2(10) NOT NULL, 
    ESTADO        VARCHAR2(10) DEFAULT 'ACTIVO' NOT NULL,
    
    CONSTRAINT CHK_USER_LONGITUD CHECK (LENGTH(USERNAME) BETWEEN 5 AND 20),
    CONSTRAINT CHK_USER_ROL CHECK (ROL IN ('ADMIN', 'USER')),
    CONSTRAINT CHK_USER_ESTADO CHECK (ESTADO IN ('ACTIVO', 'ELIMINADO'))
);

CREATE UNIQUE INDEX UIX_USERNAME_UNICO_ACTIVO ON USUARIOS (
    CASE WHEN ESTADO = 'ACTIVO' THEN USERNAME ELSE NULL END
);


-- Tabla para el Microservicio de Huéspedes
CREATE TABLE HUESPEDES (
    ID_HUESPED    NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NOMBRE        VARCHAR2(50) NOT NULL,
    APELLIDO      VARCHAR2(50) NOT NULL,
    EMAIL         VARCHAR2(100) NOT NULL,
    TELEFONO      VARCHAR2(10) NOT NULL,
    DOCUMENTO     VARCHAR2(20) NOT NULL,
    NACIONALIDAD  VARCHAR2(50) NOT NULL,
    ESTADO        VARCHAR2(10) DEFAULT 'ACTIVO' NOT NULL,
    
    
    CONSTRAINT CHK_HUESPED_NOMBRE CHECK (LENGTH(NOMBRE) >= 2),
    CONSTRAINT CHK_HUESPED_APELLIDO CHECK (LENGTH(APELLIDO) >= 2),
    CONSTRAINT CHK_HUESPED_TEL_DIGITOS CHECK (LENGTH(TELEFONO) = 10),
    CONSTRAINT CHK_HUESPED_ESTADO CHECK (ESTADO IN ('ACTIVO', 'ELIMINADO'))
);

-- Unicidad de Email
CREATE UNIQUE INDEX UIX_HUESPED_EMAIL_ACTIVO ON HUESPEDES (
    CASE WHEN ESTADO = 'ACTIVO' THEN EMAIL ELSE NULL END
);

-- Unicidad de Teléfono
CREATE UNIQUE INDEX UIX_HUESPED_TEL_ACTIVO ON HUESPEDES (
    CASE WHEN ESTADO = 'ACTIVO' THEN TELEFONO ELSE NULL END
);

-- Unicidad de Documento
CREATE UNIQUE INDEX UIX_HUESPED_DOC_ACTIVO ON HUESPEDES (
    CASE WHEN ESTADO = 'ACTIVO' THEN DOCUMENTO ELSE NULL END
);


-- Tabla para el Microservicio de Habitaciones
CREATE TABLE HABITACIONES (
    ID_HABITACION    NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NUMERO           NUMBER NOT NULL,
    TIPO             VARCHAR2(50) NOT NULL,
    PRECIO           NUMBER(10, 2) NOT NULL,
    CAPACIDAD        NUMBER NOT NULL,
    ESTADO_HABITACION    VARCHAR2(10)  DEFAULT 'DISPONIBLE'  NOT NULL, 
    ESTADO_REGISTRO  VARCHAR2(10) DEFAULT 'ACTIVO' NOT NULL, 
    
    
    CONSTRAINT CHK_HAB_NUMERO CHECK (NUMERO > 0),
    CONSTRAINT CHK_HAB_PRECIO CHECK (PRECIO > 0),
    CONSTRAINT CHK_HAB_CAPACIDAD CHECK (CAPACIDAD >= 1),
    CONSTRAINT CHK_HAB_ESTADO_REG CHECK (ESTADO_REGISTRO IN ('ACTIVO', 'ELIMINADO')),
    CONSTRAINT CHK_HAB_CATALOGO_EST CHECK (ESTADO_HABITACION IN (1, 2, 3, 4))
);

CREATE UNIQUE INDEX UIX_HAB_NUMERO_ACTIVO ON HABITACIONES (
    CASE WHEN ESTADO_REGISTRO = 'ACTIVO' THEN NUMERO ELSE NULL END
);


-- Tabla para el Microservicio de Reservas
CREATE TABLE RESERVAS (
    ID_RESERVA        NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    ID_HUESPED        NUMBER NOT NULL, 
    ID_HABITACION     NUMBER NOT NULL, 
    FECHA_ENTRADA     DATE NOT NULL,
    FECHA_SALIDA      DATE NOT NULL,
    MONTO_TOTAL       NUMBER(12, 2) NOT NULL,
    ESTADO_RESERVA     VARCHAR2(10) DEFAULT 'CONFIRMADA' NOT NULL,
    ESTADO_REGISTRO   VARCHAR2(10) DEFAULT 'ACTIVO' NOT NULL,
    
    CONSTRAINT CHK_RES_FECHAS CHECK (FECHA_ENTRADA < FECHA_SALIDA),
    CONSTRAINT CHK_RES_MONTO CHECK (MONTO_TOTAL >= 0),
    CONSTRAINT CHK_RES_CATALOGO_EST CHECK (ESTADO_RESERVA IN (1, 2, 3, 4)),
    CONSTRAINT CHK_RES_EST_REG CHECK (ESTADO_REGISTRO IN ('ACTIVO', 'ELIMINADO'))
);
